#include <iostream>
#include <string>
using namespace std;

int main() {
    int nowAge = 22;                   // 定义当前年龄（整型）
    string bornDay = "20030614";       // 出生年月日（字符串）
    bool isAdult = true;               // 是否成年（布尔型）

    cout << "今年岁数：" << nowAge << endl;
    cout << "出生年月日：" << bornDay << endl;
    cout << "是否成年：" << isAdult << endl;

    return 0;
}
